A Table can be sortable through its controlled exposed props:

```typescript jsx
import sortBy from 'lodash/sortBy';
// as fetched from the API
const items = [
  {
    id: 1,
    name: 'Bravo AWS DynamoDB Table Encryption Enabled',
    resourceType: 'AWS.DynamoDB.Table.Snapshot',
    severity: 'CRITICAL',
    status: 'Failing',
    lastModified: '28/04/2019',
  },
  {
    id: 2,
    name: 'Alpha DynamoDB Table Encryption Enabled',
    resourceType: 'AWS.DynamoDB.Table.Snapshot',
    severity: 'CRITICAL',
    status: 'Failing',
    lastModified: '28/04/2019',
  },
  {
    id: 3,
    name: 'Delta DynamoDB Table Encryption Enabled',
    resourceType: 'AWS.DynamoDB.Table.Snapshot',
    severity: 'CRITICAL',
    status: 'Failing',
    lastModified: '28/04/2019',
  },
  {
    id: 4,
    name: 'Charlie DynamoDB Table Encryption Enabled',
    resourceType: 'AWS.DynamoDB.Table.Snapshot',
    severity: 'CRITICAL',
    status: 'Failing',
    lastModified: '28/04/2019',
  },
];

const Example = () => {
  const [sortDir, setSortDir] = React.useState(null);
  const [sortKey, setSortKey] = React.useState(null);

  let sortedItems = sortBy(items, [sortKey]);
  if (sortDir === 'descending') {
    sortedItems = sortedItems.reverse();
  }
  return (
    <Card width="100%">
      <Table>
        <Table.Head>
          <Table.Row>
            <Table.SortableHeaderCell
              sortDir={sortKey === 'name' ? sortDir : false}
              onClick={() => {
                if (sortKey === 'name') {
                  setSortDir(sortDir === 'ascending' ? 'descending' : 'ascending');
                } else {
                  setSortDir('ascending');
                  setSortKey('name');
                }
              }}
            >
              Name
            </Table.SortableHeaderCell>
            <Table.HeaderCell>Resource Type</Table.HeaderCell>
            <Table.HeaderCell>Severity</Table.HeaderCell>
            <Table.HeaderCell>Status</Table.HeaderCell>
          </Table.Row>
        </Table.Head>
        <Table.Body>
          {sortedItems.map(item => (
            <Table.Row key={item.id}>
              <Table.Cell truncated>{item.name}</Table.Cell>
              <Table.Cell truncated title={item.resourceType}>
                {item.resourceType}
              </Table.Cell>
              <Table.Cell>{item.severity}</Table.Cell>
              <Table.Cell>{item.status}</Table.Cell>
            </Table.Row>
          ))}
        </Table.Body>
      </Table>
    </Card>
  );
};

<Example />;
```
