Your bread and butter Checkbox.

```typescript jsx
const Example = () => {
  const [checked, setChecked] = React.useState('A');
  return (
    <Box>
      <Box>
        <Radio
          checked={checked === 'A'}
          onChange={() => setChecked('A')}
          value="A"
          name="example"
          label="A Radio Label"
        />
      </Box>
      <Box>
        <Radio
          checked={checked === 'B'}
          onChange={() => setChecked('B')}
          value="B"
          size="large"
          name="example"
          label="A Large Radio"
        />
      </Box>
      <Box>
        <Radio
          invalid
          checked={checked === 'C'}
          onChange={() => setChecked('C')}
          value="C"
          name="example"
          label="Invalid"
        />
      </Box>
      <Box>
        <Radio
          checked={checked === 'D'}
          onChange={() => setChecked('D')}
          value="D"
          label="Disabled"
          name="example"
          disabled
        />
      </Box>
    </Box>
  );
};

<Example />;
```
